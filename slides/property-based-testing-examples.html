Property-based Testing

    
[TestMethod]
public void AddSameNumber_IsEqualToTwoTimesTheSameNumber()
{
    var calculator = new Calculator();
    Prop.ForAll(x => calculator.Add(x, x) == 2 * x);
}
    

    
[TestMethod]
public void AddZero_ResultIsEqualToNonZeroNumber()
{
    var calculator = new Calculator();
    Prop.ForAll(x => calculator.Add(x, 0) == x);
}
